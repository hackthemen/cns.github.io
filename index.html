import string

def printString(S, N):
    plaintext = [''] * 5
    freq = [0] * 26
    freqSorted = sorted(freq, reverse=True)
    used = [0] * 26
    
    S = S.upper()  # convert input to uppercase
    
    for i in range(N):
        if S[i] != " ":
            freq[ord(S[i]) - 65] += 1
    
    freqSorted = sorted(freq, reverse=True)
    
    T = "ETAOINSHRDLCUMWFGYPBVKJXQZ"
    
    for i in range(5):
        ch = -1
        for j in range(26):
            if freqSorted[i] == freq[j] and used[j] == 0:
                used[j] = 1
                ch = j
                break
        if ch == -1:
            break
        x = ch - ord(T[i]) + 65
        curr = ""
        for k in range(N):
            if S[k] not in string.ascii_uppercase:
                curr += S[k]
                continue
            y = ord(S[k]) - x
            if y < 65:
                y += 26
            curr += chr(y)
    
        plaintext[i] = curr
    
    for i in range(5):
        print(plaintext[i])

import string

def generateKey(string, key):
    key = list(key)
    if len(string) == len(key):
        return key
    else:
        for i in range(len(string) - len(key)):
            key.append(key[i % len(key)])
    return "".join(key)


def cipherText(string, key):
    cipher_text = []
    for i in range(len(string)):
        x = (ord(string[i]) + ord(key[i])) % 26
        x += ord("A")
        cipher_text.append(chr(x))
    return "".join(cipher_text)


def originalText(cipher_text, key):
    orig_text = []
    for i in range(len(cipher_text)):
        x = (ord(cipher_text[i]) - ord(key[i]) + 26) % 26
        x += ord("A")
        orig_text.append(chr(x))
    return "".join(orig_text)
